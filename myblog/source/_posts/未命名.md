```javascript
let arr = [27.2,0,'0013','14px',123];
arr = arr.map(parseInt);
console.log(arr);    //[27,NaN,1,1,27]

/*
* parseInt(27.2,0)   => 27
* parseInt(0,1)   => NaN
* parseInt('0013',2)  => 1 => 1*2^0 = 1
* parseInt('14px',3) => 1 => 1*3^0 = 1
* parseInt(123,4)  => 123 => 1*4^2+2*4^1+3*4^0 = 16+8+3=27
```
